N := 100;
@ Printing and counting all prime numbers up to: N;
n := 0; # prime counter
p := 2;
while p <= N {
    divisor := 2;
    r := sqrt p;
    while divisor <= r {
        if p % divisor = 0 {break2}
        divisor += 1
    } else { # `else` executes when the condition for `while` is not satisfied but not on `break2`
        @ prime: p;
        n += 1
    }
    p += 1
}
@ Overall prime fraction: n / N;
n
